import { usePost } from "@reflexjs/gatsby-theme-post"

export const FeaturedPosts = () => {
  const posts = usePost({ featured: true })
  return posts.length ? (
    <Grid col="1|1|2|3" gap="6|8|10">
      {posts &&
        posts.slice(0, 3).map((post, index) => (
          <Div key={index} mb="8">
            <PostTeaser {...post} />
          </Div>
        ))}
    </Grid>
  ) : null
}

export const PostTeaser = ({
  title,
  excerpt,
  slug,
  image,
  date,
  datetime,
  author,
  timeToRead,
  ...props
}) => (
  <Article {...props}>
    {image && (
      <Link href={slug} d="block">
        <Image src={image} title={title} alt={title} />
      </Link>
    )}
    {title && (
      <Link href={slug}>
        <H2 my="4" fontSize="2xl" hoverColor="primary">
          {title}
        </H2>
      </Link>
    )}
    {excerpt && <P mt="1">{excerpt}</P>}
    <PostMeta
      author={author}
      timeToRead={timeToRead}
      date={date}
      datetime={datetime}
      fontSize="md"
    />
  </Article>
)
